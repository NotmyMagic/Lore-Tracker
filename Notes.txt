Tables created already:

CREATE TABLE worlds (
    world_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE characters (
    character_id SERIAL PRIMARY KEY,
    world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    race VARCHAR(50),
    class VARCHAR(50),
    alignment VARCHAR(50),
    background TEXT,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE bbegs (
    bbeg_id SERIAL PRIMARY KEY,
    world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    title VARCHAR(100),
    species VARCHAR(50),
    goal TEXT,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE npcs (
    npc_id SERIAL PRIMARY KEY,
    world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    role VARCHAR(100),
    location VARCHAR(100),
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE factions (
    faction_id SERIAL PRIMARY KEY,
    world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    type VARCHAR(100),
    headquarters VARCHAR(100),
    goals TEXT,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE lore_entries (
	lore_id SERIAL PRIMARY KEY,
	world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,
	name VARCHAR(100) NOT NULL,
	category VARCHAR(100) NOT NULL,
	description TEXT,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE categories (
    category_id SERIAL PRIMARY KEY,
    world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,

    description TEXT

CREATE TABLE tags (
    tag_id SERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    description TEXT
);

CREATE TABLE entity_relations (
    relation_id SERIAL PRIMARY KEY,
    world_id INT NOT NULL REFERENCES worlds(world_id) ON DELETE CASCADE,
    source_type VARCHAR(50) NOT NULL,  -- 'character', 'bbeg', 'npc', 'faction'
    source_id INT NOT NULL,
    target_type VARCHAR(50) NOT NULL,  -- 'character', 'bbeg', 'npc', 'faction'
    target_id INT NOT NULL,
    tag_id INT REFERENCES tags(tag_id),
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT chk_source_type CHECK (source_type IN ('character','bbeg','npc','faction')),
    CONSTRAINT chk_target_type CHECK (target_type IN ('character','bbeg','npc','faction'))
);
